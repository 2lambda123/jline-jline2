26  /**
27   * Terminal implementation for Microsoft Windows. Terminal initialization in
28   * {@link #init} is accomplished by extracting the
29   * <em>jline_<i>version</i>.dll</em>, saving it to the system temporary
30   * directoy (determined by the setting of the <em>java.io.tmpdir</em> System
31   * property), loading the library, and then calling the Win32 APIs <a
32   * href="http://msdn.microsoft.com/library/default.asp?
33   * url=/library/en-us/dllproc/base/setconsolemode.asp">SetConsoleMode</a> and
34   * <a href="http://msdn.microsoft.com/library/default.asp?
35   * url=/library/en-us/dllproc/base/getconsolemode.asp">GetConsoleMode</a> to
36   * disable character echoing.
37   * <p/>
38   * <p>
39   * By default, the {@link #wrapInIfNeeded(java.io.InputStream)} method will attempt
40   * to test to see if the specified {@link InputStream} is {@link System#in} or a wrapper
41   * around {@link FileDescriptor#in}, and if so, will bypass the character reading to
42   * directly invoke the readc() method in the JNI library. This is so the class
43   * can read special keys (like arrow keys) which are otherwise inaccessible via
44   * the {@link System#in} stream. Using JNI reading can be bypassed by setting
45   * the <code>jline.WindowsTerminal.directConsole</code> system property
46   * to <code>false</code>.
47   * </p>
48   *
49   * @author <a href="mailto:mwp1@cornell.edu">Marc Prud'hommeaux</a>
50   * @author <a href="mailto:jason@planet57.com">Jason Dillon</a>
51   * @since 2.0
52   */
53  public class WindowsTerminal
54      extends TerminalSupport

56      public static final String DIRECT_CONSOLE = WindowsTerminal.class.getName() + ".directConsole";
57  
58      public static final String ANSI = WindowsTerminal.class.getName() + ".ansi";

64      public WindowsTerminal() throws Exception {
65          super(true);

68      @Override
69      public void init() throws Exception {
70          super.init();
71  
72          setAnsiSupported(Configuration.getBoolean(ANSI, true));
73  
74          //
75          // FIXME: Need a way to disable direct console and sysin detection muck
76          //
77  
78          setDirectConsole(Configuration.getBoolean(DIRECT_CONSOLE, true));
79  
80          this.originalMode = getConsoleMode();
81          setConsoleMode(originalMode & ~ENABLE_ECHO_INPUT.code);
82          setEchoEnabled(false);
83      }

85      /**
86       * Restore the original terminal configuration, which can be used when
87       * shutting down the console reader. The ConsoleReader cannot be
88       * used after calling this method.
89       */
90      @Override
91      public void restore() throws Exception {
92          // restore the old console mode
93          setConsoleMode(originalMode);
94          super.restore();
95      }

109     @Override
110     public void setEchoEnabled(final boolean enabled) {
111         // Must set these four modes at the same time to make it work fine.
112         if (enabled) {
113             setConsoleMode(getConsoleMode() |
114                 ENABLE_ECHO_INPUT.code |
115                 ENABLE_LINE_INPUT.code |
116                 ENABLE_PROCESSED_INPUT.code |
117                 ENABLE_WINDOW_INPUT.code);
118         }
119         else {
120             setConsoleMode(getConsoleMode() &
121                 ~(ENABLE_LINE_INPUT.code |
122                     ENABLE_ECHO_INPUT.code |
123                     ENABLE_PROCESSED_INPUT.code |
124                     ENABLE_WINDOW_INPUT.code));
125         }
126         super.setEchoEnabled(enabled);
127     }

138      * Whether or not to allow the use of the JNI console interaction.

145     @Override
146     public InputStream wrapInIfNeeded(InputStream in) throws IOException {
147         if (directConsole && isSystemIn(in)) {
148             return new InputStream() {
149                 @Override
150                 public int read() throws IOException {
151                     return readByte();
152                 }
153             };
154         } else {
155             return super.wrapInIfNeeded(in);
156         }
157     }
158 
159     protected boolean isSystemIn(final InputStream in) throws IOException {
160         if (in == null) {
161             return false;
162         }
163         else if (in == System.in) {
164             return true;
165         }
166         else if (in instanceof FileInputStream && ((FileInputStream) in).getFD() == FileDescriptor.in) {
167             return true;
168         }
169 
170         return false;
171     }

196     /**
197      * Console mode
198      * <p/>
199      * Constants copied <tt>wincon.h</tt>.
200      */
201     public static enum ConsoleMode
202     {
203         /**
204          * The ReadFile or ReadConsole function returns only when a carriage return
205          * character is read. If this mode is disable, the functions return when one
206          * or more characters are available.
207          */
208         ENABLE_LINE_INPUT(2),
209 
210         /**
211          * Characters read by the ReadFile or ReadConsole function are written to
212          * the active screen buffer as they are read. This mode can be used only if
213          * the ENABLE_LINE_INPUT mode is also enabled.
214          */
215         ENABLE_ECHO_INPUT(4),
216 
217         /**
218          * CTRL+C is processed by the system and is not placed in the input buffer.
219          * If the input buffer is being read by ReadFile or ReadConsole, other
220          * control keys are processed by the system and are not returned in the
221          * ReadFile or ReadConsole buffer. If the ENABLE_LINE_INPUT mode is also
222          * enabled, backspace, carriage return, and linefeed characters are handled
223          * by the system.
224          */
225         ENABLE_PROCESSED_INPUT(1),
226 
227         /**
228          * User interactions that change the size of the console screen buffer are
229          * reported in the console's input buffee. Information about these events
230          * can be read from the input buffer by applications using
231          * theReadConsoleInput function, but not by those using ReadFile
232          * orReadConsole.
233          */
234         ENABLE_WINDOW_INPUT(8),
235 
236         /**
237          * If the mouse pointer is within the borders of the console window and the
238          * window has the keyboard focus, mouse events generated by mouse movement
239          * and button presses are placed in the input buffer. These events are
240          * discarded by ReadFile or ReadConsole, even when this mode is enabled.
241          */
242         ENABLE_MOUSE_INPUT(16),
243 
244         /**
245          * When enabled, text entered in a console window will be inserted at the
246          * current cursor location and all text following that location will not be
247          * overwritten. When disabled, all following text will be overwritten. An OR
248          * operation must be performed with this flag and the ENABLE_EXTENDED_FLAGS
249          * flag to enable this functionality.
250          */
251         ENABLE_PROCESSED_OUTPUT(1),
252 
253         /**
254          * This flag enables the user to use the mouse to select and edit text. To
255          * enable this option, use the OR to combine this flag with
256          * ENABLE_EXTENDED_FLAGS.
257          */
258         ENABLE_WRAP_AT_EOL_OUTPUT(2),;
259 
260         public final int code;
261 
262         ConsoleMode(final int code) {
263             this.code = code;
264         }
265     }
